### الحل 3: تحديث نظام البحث المتقدم ✅

#### إنشاء خدمة بحث متقدمة جديدة

```typescript
// src/services/advancedSearchService.ts

import { 
  collection, 
  query, 
  where, 
  orderBy, 
  limit,
  getDocs,
  Query,
  DocumentData
} from 'firebase/firestore';
import { db } from '../firebase/firebase-config';
import { SearchData } from '../pages/AdvancedSearchPage/types';
import { CarListing } from '../types/CarListing';

class AdvancedSearchService {
  private collectionName = 'cars';

  /**
   * Build Firestore query from search data
   */
  private buildQuery(searchData: SearchData): Query<DocumentData> {
    let q = query(collection(db, this.collectionName));
    
    // Basic filters
    if (searchData.make) {
      q = query(q, where('make', '==', searchData.make));
    }
    
    if (searchData.model) {
      q = query(q, where('model', '==', searchData.model));
    }
    
    if (searchData.vehicleType) {
      q = query(q, where('vehicleType', '==', searchData.vehicleType));
    }
    
    if (searchData.fuelType) {
      q = query(q, where('fuelType', '==', searchData.fuelType));
    }
    
    if (searchData.transmission) {
      q = query(q, where('transmission', '==', searchData.transmission));
    }
    
    if (searchData.condition) {
      q = query(q, where('condition', '==', searchData.condition));
    }
    
    if (searchData.sellerType) {
      q = query(q, where('sellerType', '==', searchData.sellerType));
    }
    
    // Location filters
    if (searchData.city) {
      q = query(q, where('city', '==', searchData.city));
    }
    
    if (searchData.country) {
      q = query(q, where('country', '==', searchData.country));
    }
    
    // Range filters (will be applied client-side)
    // Firestore doesn't support multiple range queries
    
    // Sorting
    q = query(q, orderBy('createdAt', 'desc'));
    
    // Limit results
    q = query(q, limit(100));
    
    return q;
  }

  /**
   * Apply client-side filters for ranges
   */
  private applyClientFilters(
    cars: CarListing[], 
    searchData: SearchData
  ): CarListing[] {
    return cars.filter(car => {
      // Price range
      if (searchData.priceFrom && car.price < parseFloat(searchData.priceFrom)) {
        return false;
      }
      if (searchData.priceTo && car.price > parseFloat(searchData.priceTo)) {
        return false;
      }
      
      // Year range
      if (searchData.firstRegistrationFrom) {
        const yearFrom = parseInt(searchData.firstRegistrationFrom);
        if (car.year < yearFrom) return false;
      }
      if (searchData.firstRegistrationTo) {
        const yearTo = parseInt(searchData.firstRegistrationTo);
        if (car.year > yearTo) return false;
      }
      
      // Mileage range
      if (searchData.mileageFrom && car.mileage < parseFloat(searchData.mileageFrom)) {
        return false;
      }
      if (searchData.mileageTo && car.mileage > parseFloat(searchData.mileageTo)) {
        return false;
      }
      
      // Power range
      if (searchData.powerFrom && (!car.power || car.power < parseFloat(searchData.powerFrom))) {
        return false;
      }
      if (searchData.powerTo && (!car.power || car.power > parseFloat(searchData.powerTo))) {
        return false;
      }
      
      // Engine size range
      if (searchData.cubicCapacityFrom) {
        const ccFrom = parseFloat(searchData.cubicCapacityFrom);
        if (!car.engineSize || car.engineSize < ccFrom) return false;
      }
      if (searchData.cubicCapacityTo) {
        const ccTo = parseFloat(searchData.cubicCapacityTo);
        if (!car.engineSize || car.engineSize > ccTo) return false;
      }
      
      // Seats range
      if (searchData.seatsFrom) {
        const seatsFrom = parseInt(searchData.seatsFrom);
        if (!(car as any).numberOfSeats || (car as any).numberOfSeats < seatsFrom) {
          return false;
        }
      }
      if (searchData.seatsTo) {
        const seatsTo = parseInt(searchData.seatsTo);
        if (!(car as any).numberOfSeats || (car as any).numberOfSeats > seatsTo) {
          return false;
        }
      }
      
      // Doors range
      if (searchData.doorsFrom) {
        const doorsFrom = parseInt(searchData.doorsFrom);
        if (!(car as any).numberOfDoors || (car as any).numberOfDoors < doorsFrom) {
          return false;
        }
      }
      if (searchData.doorsTo) {
        const doorsTo = parseInt(searchData.doorsTo);
        if (!(car as any).numberOfDoors || (car as any).numberOfDoors > doorsTo) {
          return false;
        }
      }
      
      // Color filters
      if (searchData.exteriorColor && car.color !== searchData.exteriorColor) {
        return false;
      }
      
      if (searchData.interiorColor) {
        if (!(car as any).interiorColor || (car as any).interiorColor !== searchData.interiorColor) {
          return false;
        }
      }
      
      if (searchData.interiorMaterial) {
        if (!(car as any).interiorMaterial || (car as any).interiorMaterial !== searchData.interiorMaterial) {
          return false;
        }
      }
      
      // Boolean filters
      if (searchData.adsWithPictures && (!car.images || car.images.length === 0)) {
        return false;
      }
      
      if (searchData.adsWithVideo && !(car as any).hasVideo) {
        return false;
      }
      
      if (searchData.nonSmokerVehicle && !(car as any).nonSmoker) {
        return false;
      }
      
      if (searchData.warranty && !(car as any).warranty) {
        return false;
      }
      
      // Description search
      if (searchData.searchDescription) {
        const searchTerm = searchData.searchDescription.toLowerCase();
        const description = car.description?.toLowerCase() || '';
        if (!description.includes(searchTerm)) {
          return false;
        }
      }
      
      return true;
    });
  }

  /**
   * Search cars with advanced filters
   */
  async searchCars(searchData: SearchData): Promise<CarListing[]> {
    try {
      console.log('🔍 Advanced search with filters:', searchData);
      
      // Build and execute Firestore query
      const q = this.buildQuery(searchData);
      const querySnapshot = await getDocs(q);
      
      // Convert to CarListing array
      const cars: CarListing[] = [];
      querySnapshot.forEach((doc) => {
        const data = doc.data();
        cars.push({
          id: doc.id,
          ...data,
          createdAt: data.createdAt?.toDate(),
          updatedAt: data.updatedAt?.toDate(),
          expiresAt: data.expiresAt?.toDate()
        } as CarListing);
      });
      
      console.log(`✅ Found ${cars.length} cars from Firestore`);
      
      // Apply client-side filters
      const filteredCars = this.applyClientFilters(cars, searchData);
      
      console.log(`✅ After client filters: ${filteredCars.length} cars`);
      
      return filteredCars;
    } catch (error) {
      console.error('❌ Error in advanced search:', error);
      throw error;
    }
  }
}

export default new AdvancedSearchService();
```
