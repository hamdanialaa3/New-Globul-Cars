#### الوظائف الرئيسية

##### 1. حفظ الحالة
```typescript
static saveState(data: Record<string, any>, currentStep: string): void {
  const state: WorkflowState = {
    data,
    images: this.getImages(),
    lastUpdated: Date.now(),
    currentStep
  };
  localStorage.setItem(STORAGE_KEY, JSON.stringify(state));
}
```

##### 2. تحميل الحالة
```typescript
static loadState(): WorkflowState | null {
  const saved = localStorage.getItem(STORAGE_KEY);
  if (!saved) return null;

  const state: WorkflowState = JSON.parse(saved);

  // التحقق من انتهاء الصلاحية (24 ساعة)
  if (Date.now() - state.lastUpdated > EXPIRY_TIME) {
    this.clearState();
    return null;
  }

  return state;
}
```

##### 3. حفظ الصور
```typescript
static async saveImages(files: File[]): Promise<void> {
  const base64Images = await Promise.all(
    files.map(file => ImageOptimizationService.imageToBase64(file))
  );
  localStorage.setItem(IMAGES_KEY, JSON.stringify(base64Images));
}
```

##### 4. الحفظ التلقائي
```typescript
static autoSave(data: Record<string, any>, currentStep: string): void {
  // Debounce: انتظار 500ms قبل الحفظ
  const timeout = setTimeout(() => {
    this.saveState(data, currentStep);
  }, 500);
  
  (window as any).__workflowAutoSaveTimeout = timeout;
}
```

## 🔄 تدفق البيانات (Data Flow)

### المسار الكامل

```
1. المستخدم يختار نوع السيارة
   ↓
   URL: /sell/auto → /sell/inserat/car/verkaeufertyp?vt=car

2. المستخدم يختار نوع البائع
   ↓
   URL: ?vt=car&st=private

3. المستخدم يدخل بيانات السيارة
   ↓
   URL: ?vt=car&st=private&mk=BMW&md=X5&fy=2020&mi=50000

4. المستخدم يختار معدات السلامة
   ↓
   URL: ?...&safety=abs,esp,airbags

5. المستخدم يختار معدات الراحة
   ↓
   URL: ?...&comfort=ac,heatedSeats

6. المستخدم يختار الترفيه
   ↓
   URL: ?...&infotainment=bluetooth,navigation

7. المستخدم يختار الإضافات
   ↓
   URL: ?...&extras=sunroof,leatherSeats

8. المستخدم يرفع الصور
   ↓
   الصور تُحفظ في localStorage مؤقتاً

9. المستخدم يحدد السعر
   ↓
   URL: ?...&pr=15000&ng=true

10. المستخدم يدخل بيانات الاتصال
    ↓
    URL: ?...&name=John&email=john@example.com&phone=+359...

11. النشر النهائي
    ↓
    sellWorkflowService.createCarListing()
    ↓
    Firebase Firestore + Storage
    ↓
    إعلان جديد تم إنشاؤه ✅
```

```typescript
  const requiredFields = [
    { key: 'make', label: 'Make (Марка)' },
    { key: 'model', label: 'Model (Модел)' },
    { key: 'year', label: 'Year (Година)' },
    { key: 'price', label: 'Price (Цена)' },
    { key: 'sellerName', label: 'Seller Name (Име)' },
    { key: 'sellerEmail', label: 'Seller Email (Имейл)' },
    { key: 'sellerPhone', label: 'Seller Phone (Телефон)' }
  ];

  const missingFields: string[] = [];
  
  for (const field of requiredFields) {
    if (!workflowData[field.key]) {
      missingFields.push(field.label);
    }
  }

  if (!workflowData.region && !workflowData.city) {
    missingFields.push('Location (Местоположение)');
  }

  return {
    isValid: missingFields.length === 0,
    missingFields
  };
}
```

```typescript
static async compressImage(file: File, maxSize: number): Promise<File> {
  // 1. قراءة الصورة
  const image = await this.loadImage(file);
  
  // 2. إنشاء canvas
  const canvas = document.createElement('canvas');
  const ctx = canvas.getContext('2d');
  
  // 3. تغيير الحجم
  canvas.width = image.width;
  canvas.height = image.height;
  ctx.drawImage(image, 0, 0);
  
  // 4. ضغط الصورة
  const blob = await new Promise<Blob>((resolve) => {
    canvas.toBlob((blob) => resolve(blob!), 'image/jpeg', 0.8);
  });
  
  // 5. إرجاع File جديد
  return new File([blob], file.name, { type: 'image/jpeg' });
}
```

```typescript
class CarBrandsCache {
  private static cache = new Map<string, any>();
  private static TTL = 24 * 60 * 60 * 1000; // 24 hours

  static get(key: string): any | null {
    const cached = this.cache.get(key);
    if (!cached) return null;
    
    if (Date.now() - cached.timestamp > this.TTL) {
      this.cache.delete(key);
      return null;
    }
    
    return cached.data;
  }

  static set(key: string, data: any): void {
    this.cache.set(key, {
      data,
      timestamp: Date.now()
    });
  }
}
```
