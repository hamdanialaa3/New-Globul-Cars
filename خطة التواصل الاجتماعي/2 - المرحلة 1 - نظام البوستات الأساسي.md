const translateText = async (text: string, from: string, to: string) => {
  const response = await fetch(
    `https://translation.googleapis.com/language/translate/v2?key=${API_KEY}`,
    {
      method: 'POST',
      body: JSON.stringify({
        q: text,
        source: from,
        target: to,
        format: 'text'
      })
    }
  );
  
  const data = await response.json();
  return data.data.translations[0].translatedText;
};

## 1.7 الكود الجاهز للتنفيذ

### CreatePostForm - Structure

```typescript
import React, { useState } from 'react';
import { useAuth } from '../../contexts/AuthProvider';
import { useLanguage } from '../../contexts/LanguageContext';
import postsService from '../../services/social/posts.service';

const CreatePostForm: React.FC<{ onClose: () => void }> = ({ onClose }) => {
  const { user } = useAuth();
  const { language, t } = useLanguage();
  
  // State
  const [postType, setPostType] = useState<PostType>('text');
  const [text, setText] = useState('');
  const [media, setMedia] = useState<File[]>([]);
  const [selectedCar, setSelectedCar] = useState<Car | null>(null);
  const [visibility, setVisibility] = useState<Visibility>('public');
  const [location, setLocation] = useState({ city: '', region: '' });
  const [hashtags, setHashtags] = useState<string[]>([]);
  const [mentions, setMentions] = useState<string[]>([]);
  const [loading, setLoading] = useState(false);
  
  // Auto-detect hashtags
  useEffect(() => {
    const detected = text.match(/#[\w\u0400-\u04FF]+/g) || [];
    setHashtags(detected.map(h => h.slice(1)));
  }, [text]);
  
  // Auto-detect mentions
  useEffect(() => {
    const detected = text.match(/@[\w\u0400-\u04FF]+/g) || [];
    setMentions(detected.map(m => m.slice(1)));
  }, [text]);
  
  // Submit handler
  const handleSubmit = async () => {
    // Validation
    if (!validatePost()) return;
    
    setLoading(true);
    
    try {
      // 1. Upload media
      const mediaUrls = await uploadMedia();
      
      // 2. Create post
      const postData = {
        type: postType,
        content: {
          text,
          media: mediaUrls.length > 0 ? {
            type: media[0].type.startsWith('video') ? 'video' : 'image',
            urls: mediaUrls
          } : undefined,
          carReference: selectedCar ? {
            carId: selectedCar.id,
            carTitle: `${selectedCar.brand} ${selectedCar.model}`,
            carImage: selectedCar.images[0]
          } : undefined,
          hashtags
        },
        visibility,
        location: location.city ? location : undefined
      };
      
      const newPost = await postsService.createPost(postData);
      
      // 3. Close modal
      onClose();
      
      // 4. Show success
      showToast(t('post.created.success'));
      
    } catch (error) {
      showToast(t('post.created.error'), 'error');
    } finally {
      setLoading(false);
    }
  };
  
  return (
    <FormContainer>
      {/* Header */}
      <FormHeader>
        <h3>{t('post.create.title')}</h3>
        <CloseButton onClick={onClose} />
      </FormHeader>
      
      {/* Post Type Selector */}
      <PostTypeSelector 
        selected={postType}
        onChange={setPostType}
      />
      
      {/* Text Editor */}
      <RichTextEditor
        value={text}
        onChange={setText}
        placeholder={t('post.create.placeholder')}
        maxLength={5000}
      />
      
      {/* Media Upload */}
      <MediaUploader
        files={media}
        onChange={setMedia}
        maxFiles={10}
        maxSize={5 * 1024 * 1024}
      />
      
      {/* Car Selector */}
      {postType === 'car_showcase' && (
        <CarSelector
          selected={selectedCar}
          onChange={setSelectedCar}
        />
      )}
      
      {/* Options */}
      <PostOptions>
        <VisibilitySelector 
          value={visibility}
          onChange={setVisibility}
        />
        <LocationSelector
          value={location}
          onChange={setLocation}
        />
      </PostOptions>
      
      {/* Footer */}
      <FormFooter>
        <CharacterCount>
          {text.length}/5000
        </CharacterCount>
        <SubmitButton 
          onClick={handleSubmit}
          disabled={!canSubmit() || loading}
        >
          {loading ? t('post.creating') : t('post.create')}
        </SubmitButton>
      </FormFooter>
    </FormContainer>
  );
};
```
