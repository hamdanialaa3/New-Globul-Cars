**الملف:** `src/services/social/post-scoring.service.ts` (250 سطر)

```typescript
interface PostScore {
  totalScore: number;      // المجموع الكلي (0-100)
  engagementScore: number; // التفاعل (0-40)
  recencyScore: number;    // الحداثة (0-20)
  qualityScore: number;    // الجودة (0-20)
  authorScore: number;     // المؤلف (0-20)
}
```

**1. Engagement Score (40 نقطة):**
```
- Reactions: عدد × 1 نقطة
- Comments: عدد × 3 نقاط
- Shares: عدد × 5 نقاط
- Saves: عدد × 4 نقاط
- Views: (عدد / 100) نقطة

الحد الأقصى: 40 نقطة
```

**2. Recency Score (20 نقطة):**
```
- آخر ساعة: 20 نقطة
- آخر 3 ساعات: 18 نقطة
- آخر 6 ساعات: 15 نقطة
- آخر 12 ساعة: 12 نقطة
- آخر 24 ساعة: 8 نقاط
- آخر 3 أيام: 4 نقاط
- أكثر من 3 أيام: 1 نقطة
```

**3. Quality Score (20 نقطة):**
```
- يحتوي صور: +5 نقاط
- يحتوي فيديو: +7 نقاط
- مربوط بسيارة: +5 نقاط
- نص طويل (>200 حرف): +3 نقاط
- يحتوي hashtags: +2 نقاط
- لا تقارير spam: +3 نقاط
```

**4. Author Score (20 نقطة):**
```
- Trust Score المؤلف: (0-10) نقاط
- عدد Followers: (count / 100) نقاط (max 5)
- Profile verified: +5 نقاط
- Profile type:
  - Company: +3 نقاط
  - Dealer: +2 نقطة
  - Private: 0 نقاط
```

### 2.2 خوارزمية الترتيب (Ranking Algorithm)

**الملف:** `src/services/social/feed-algorithm.service.ts` (280 سطر)

**عملية الترتيب:**

```typescript
// 1. جلب البوستات المرشحة
async getCandidatePosts(userId: string, limit: number = 100) {
  // A. بوستات المتابَعين (50%)
  const followingPosts = await getFollowingPosts(userId, limit / 2);
  
  // B. بوستات شعبية عامة (30%)
  const trendingPosts = await getTrendingPosts(limit * 0.3);
  
  // C. بوستات من نفس المنطقة (20%)
  const localPosts = await getLocalPosts(userCity, limit * 0.2);
  
  return [...followingPosts, ...trendingPosts, ...localPosts];
}

// 2. حساب النقاط لكل بوست
async scoreAllPosts(posts: Post[], userId: string) {
  return Promise.all(
    posts.map(async (post) => {
      const score = await calculatePostScore(post, userId);
      return { post, score };
    })
  );
}

// 3. ترتيب حسب النقاط
async rankPosts(scoredPosts: ScoredPost[]) {
  return scoredPosts.sort((a, b) => b.score.totalScore - a.score.totalScore);
}

// 4. تنويع النتائج (Diversity)
async diversifyFeed(rankedPosts: Post[], userId: string) {
  const diversified = [];
  const usedAuthors = new Set<string>();
  const usedTypes = new Map<string, number>();
  
  for (const post of rankedPosts) {
    // تجنب تكرار نفس المؤلف متتالياً
    if (usedAuthors.has(post.authorId)) {
      continue;
    }
    
    // تنويع أنواع البوستات
    const typeCount = usedTypes.get(post.type) || 0;
    if (typeCount >= 2) continue;
    
    diversified.push(post);
    usedAuthors.add(post.authorId);
    usedTypes.set(post.type, typeCount + 1);
    
    // Reset every 5 posts
    if (diversified.length % 5 === 0) {
      usedAuthors.clear();
    }
  }
  
  return diversified;
```

```typescript
function calculatePostSimilarity(post1: Post, post2: Post) {
  const f1 = extractPostFeatures(post1);
  const f2 = extractPostFeatures(post2);
  
  let similarity = 0;
  
  // تطابق العلامة التجارية (وزن 30%)
  if (f1.carBrand === f2.carBrand && f1.carBrand) {
    similarity += 0.3;
  }
  
  // تطابق الموقع (وزن 20%)
  if (f1.location === f2.location && f1.location) {
    similarity += 0.2;
  }
  
  // تطابق النوع (وزن 15%)
  if (f1.type === f2.type) {
    similarity += 0.15;
  }
  
  // Hashtags المشتركة (وزن 25%)
  const commonHashtags = f1.hashtags.filter(h => f2.hashtags.includes(h));
  similarity += (commonHashtags.length / Math.max(f1.hashtags.length, 1)) * 0.25;
  
  // وجود ميديا (وزن 10%)
  if (f1.hasMedia === f2.hasMedia) {
    similarity += 0.1;
  }
  
  return similarity;
}

// التوصية بناءً على المحتوى
async recommendSimilarContent(userId: string) {
  const userLikedPosts = await getEngagedPosts(userId, 'like');
  const candidatePosts = await getRecentPosts(100);

    const similarities = userLikedPosts.map(liked => 
      calculatePostSimilarity(liked, candidate)
    );
    const avgSimilarity = similarities.reduce((a, b) => a + b, 0) / similarities.length;
    
    return { post: candidate, score: avgSimilarity };
  });
```

```typescript
export function useInfiniteScroll(
  loadMore: () => Promise<void>,
  hasMore: boolean
) {
  const [loading, setLoading] = useState(false);
  const observerRef = useRef<IntersectionObserver>();
  const sentinelRef = useRef<HTMLDivElement>(null);
  
  useEffect(() => {
    const options = {
      root: null,
      rootMargin: '200px', // تحميل قبل الوصول بـ 200px
      threshold: 0.1
    };
    
    observerRef.current = new IntersectionObserver(async ([entry]) => {
      if (entry.isIntersecting && hasMore && !loading) {
        setLoading(true);
        await loadMore();
        setLoading(false);
      }
    }, options);
    
    if (sentinelRef.current) {
      observerRef.current.observe(sentinelRef.current);
    }
    
    return () => observerRef.current?.disconnect();
  }, [loadMore, hasMore, loading]);
  
  return { sentinelRef, loading };
}
```

```typescript
    type: 'rerank' | 'inject' | 'remove';
    targetPost: string;
    newPosition: number;
  }
  
  // المكافأة (Reward)
  reward: {
    immediate: {
      engagement: number;     // تفاعل فوري (+1 إلى +10)
      timeSpent: number;      // وقت القراءة (+1 إلى +5)
    };
    delayed: {
      retention: number;      // عودة المستخدم (+5 إلى +20)
      satisfaction: number;   // رضا عام (+1 إلى +10)
    };
    total: number;           // المجموع
  }
  
  // السياسة (Policy)
  policy: {
    exploration: 0.1;        // استكشاف 10%
    exploitation: 0.9;       // استغلال 90%
    updateFrequency: 3600;   // تحديث كل ساعة
  }
}

// Agent Training
class RLAgent {
  async selectAction(state: State): Promise<Action> {
    // Epsilon-greedy strategy
    if (Math.random() < this.policy.exploration) {
      return this.randomAction(); // استكشاف
    } else {
      return this.bestAction(state); // استغلال
    }
  }
  
  async updatePolicy(experience: Experience) {
    // Q-Learning update
    const qValue = await this.getQValue(experience.state, experience.action);
    const maxFutureQ = await this.getMaxQValue(experience.nextState);
    
    const newQ = qValue + this.learningRate * (
      experience.reward + this.discountFactor * maxFutureQ - qValue
    );
    
    await this.updateQValue(experience.state, experience.action, newQ);
  }
}
```
