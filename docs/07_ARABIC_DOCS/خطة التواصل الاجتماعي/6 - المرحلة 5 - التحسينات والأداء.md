## 5.1 Caching Strategy

### الملف: `src/services/social/cache/feed-cache.service.ts` (200 سطر)

**3 مستويات Cache:**

```typescript
class FeedCacheService {
  // Layer 1: Memory (1ms)
  private memoryCache = new Map<string, CachedFeed>();
  
  // Layer 2: IndexedDB (5-10ms)
  private idbName = 'social-feed-cache';
  
  // Layer 3: Firestore (50ms)
  private firestoreCache = true;
  
  async getFeed(userId: string): Promise<Post[] | null> {
    // 1. Memory
    const memory = this.memoryCache.get(userId);
    if (memory && !this.isExpired(memory)) {
      return memory.posts;
    }
    
    // 2. IndexedDB
    const idb = await this.getFromIndexedDB(userId);
    if (idb && !this.isExpired(idb)) {
      this.memoryCache.set(userId, idb);
      return idb.posts;
    }
    
    // 3. Firestore
    const firestore = await this.getFromFirestore(userId);
    if (firestore) {
      await this.saveToAllLayers(userId, firestore);
      return firestore.posts;
    }
    
    return null;
  }
}
```

## 5.2 Real-time Updates

### الملف: `src/services/social/realtime/real-time-feed.service.ts` (230 سطر)

```typescript
class RealTimeFeedService {
  subscribeToUpdates(
    userId: string,
    callback: (update: FeedUpdate) => void
  ) {
    const feedRef = collection(db, 'feeds', userId, 'personalFeed');
    
    return onSnapshot(
      query(feedRef, orderBy('addedAt', 'desc'), limit(20)),
      (snapshot) => {
        snapshot.docChanges().forEach(change => {
          if (change.type === 'added') {
            callback({ type: 'new_post', post: change.doc.data() });
          } else if (change.type === 'modified') {
            callback({ type: 'post_updated', post: change.doc.data() });
          } else if (change.type === 'removed') {
            callback({ type: 'post_removed', postId: change.doc.id });
          }
        });
      }
    );
  }
}
```
