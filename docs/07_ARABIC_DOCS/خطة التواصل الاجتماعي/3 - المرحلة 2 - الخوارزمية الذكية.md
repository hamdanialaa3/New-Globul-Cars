```typescript
interface PostScore {
  totalScore: number;      // 0-100
  engagementScore: number; // 0-40
  recencyScore: number;    // 0-20
  qualityScore: number;    // 0-20
  authorScore: number;     // 0-20
  breakdown: ScoreBreakdown;
}

class PostScoringService {
  // حساب النقاط الكلية
  async calculatePostScore(postId: string): Promise<PostScore> {
    const post = await this.getPost(postId);
    const engagement = await this.getEngagement(postId);
    const author = await this.getAuthor(post.authorId);
    
    const scores = {
      engagement: this.calculateEngagementScore(engagement),
      recency: this.calculateRecencyScore(post.createdAt),
      quality: this.calculateQualityScore(post),
      author: this.calculateAuthorScore(author)
    };
    
    const total = Object.values(scores).reduce((a, b) => a + b, 0);
    
    return {
      totalScore: Math.min(total, 100),
      engagementScore: scores.engagement,
      recencyScore: scores.recency,
      qualityScore: scores.quality,
      authorScore: scores.author,
      breakdown: this.getBreakdown(scores)
    };
  }
  
  // 1. Engagement Score (40 نقطة)
  private calculateEngagementScore(engagement: Engagement): number {
    const weights = {
      reaction: 1,
      comment: 3,
      share: 5,
      save: 4,
      view: 0.01
    };
    
    const rawScore = 
      (engagement.reactions * weights.reaction) +
      (engagement.comments * weights.comment) +
      (engagement.shares * weights.share) +
      (engagement.saves * weights.save) +
      (engagement.views * weights.view);
    
    // Normalize to 0-40
    return Math.min(rawScore / 10, 40);
  }
  
  // 2. Recency Score (20 نقطة)
  private calculateRecencyScore(createdAt: Timestamp): number {
    const ageHours = (Date.now() - createdAt.toMillis()) / 3600000;
    
    if (ageHours < 1) return 20;
    if (ageHours < 3) return 18;
    if (ageHours < 6) return 15;
    if (ageHours < 12) return 12;
    if (ageHours < 24) return 8;
    if (ageHours < 72) return 4;
    return 1;
  }
  
  // 3. Quality Score (20 نقطة)
  private calculateQualityScore(post: Post): number {
    let score = 0;
    
    // يحتوي صور
    if (post.content.media?.type === 'image') {
      score += 5;
    }
    
    // يحتوي فيديو
    if (post.content.media?.type === 'video') {
      score += 7;
    }
    
    // مربوط بسيارة
    if (post.content.carReference) {
      score += 5;
    }
    
    // نص طويل (> 200 حرف)
    if (post.content.text.length > 200) {
      score += 3;
    }
    
    // يحتوي hashtags
    if (post.content.hashtags && post.content.hashtags.length > 0) {
      score += 2;
    }
    
    // لا تقارير spam
    const reports = await this.getReports(post.id);
    if (reports === 0) {
      score += 3;
    } else {
      score -= reports * 2; // عقوبة
    }
    
    return Math.max(0, Math.min(score, 20));
  }
  
  // 4. Author Score (20 نقطة)
  private calculateAuthorScore(author: User): number {
    let score = 0;
    
    // Trust Score (0-10)
    score += Math.min(author.trustScore / 10, 10);
    
    // Followers count
    score += Math.min(author.stats.followers / 100, 5);
    
    // Verified account
    if (author.isVerified) {
      score += 5;
    }
    
    // Profile type
    if (author.profileType === 'company') {
      score += 3;
    } else if (author.profileType === 'dealer') {
      score += 2;
    }
    
    return Math.min(score, 20);
  }
}

export default new PostScoringService();
```

```typescript
class FeedAlgorithmService {
  // جلب البوستات المرشحة
  async getCandidatePosts(
    userId: string, 
    limit: number = 100
  ): Promise<Post[]> {
    const user = await this.getUser(userId);
    
    // A. بوستات المتابَعين (50%)
    const followingPosts = await this.getFollowingPosts(
      userId, 
      Math.floor(limit * 0.5)
    );
    
    // B. بوستات شعبية عامة (30%)
    const trendingPosts = await this.getTrendingPosts(
      Math.floor(limit * 0.3)
    );
    
    // C. بوستات من نفس المنطقة (20%)
    const localPosts = await this.getLocalPosts(
      user.location?.city, 
      Math.floor(limit * 0.2)
    );
    
    // دمج وإزالة المكررات
    const candidates = this.deduplicatePosts([
      ...followingPosts,
      ...trendingPosts,
      ...localPosts
    ]);
    
    return candidates;
  }
  
  // حساب النقاط لجميع البوستات
  async scoreAllPosts(
    posts: Post[], 
    userId: string
  ): Promise<ScoredPost[]> {
    const scoredPosts = await Promise.all(
      posts.map(async (post) => {
        const score = await postScoringService.calculatePostScore(post.id);
        const personalizedScore = await this.personalizeScore(post, userId, score);
        
        return {
          post,
          score: personalizedScore
        };
      })
    );
    
    return scoredPosts;
  }
  
  // ترتيب حسب النقاط
  rankPosts(scoredPosts: ScoredPost[]): Post[] {
    return scoredPosts
      .sort((a, b) => b.score.totalScore - a.score.totalScore)
      .map(sp => sp.post);
  }
  
  // تنويع النتائج (Diversity)
  async diversifyFeed(
    rankedPosts: Post[], 
    userId: string
  ): Promise<Post[]> {
    const diversified: Post[] = [];
    const usedAuthors = new Set<string>();
    const usedTypes = new Map<string, number>();
    
    for (const post of rankedPosts) {
      // تجنب تكرار نفس المؤلف متتالياً
      if (diversified.length > 0) {
        const lastPost = diversified[diversified.length - 1];
        if (lastPost.authorId === post.authorId) {
          continue; // تخطي
        }
      }
      
      // تنويع أنواع البوستات
      const typeCount = usedTypes.get(post.type) || 0;
      if (typeCount >= 2) {
        continue; // تخطي - نوع متكرر
      }
      
      diversified.push(post);
      usedAuthors.add(post.authorId);
      usedTypes.set(post.type, typeCount + 1);
      
      // Reset كل 5 posts
      if (diversified.length % 5 === 0) {
        usedAuthors.clear();
        usedTypes.clear();
      }
    }
    
    return diversified;
  }
  
  // الدالة الرئيسية
  async getPersonalizedFeed(
    userId: string,
    page: number = 1,
    pageSize: number = 10
  ): Promise<Post[]> {
    // 1. جلب المرشحين
    const candidates = await this.getCandidatePosts(userId, 100);
    
    // 2. حساب النقاط
    const scored = await this.scoreAllPosts(candidates, userId);
    
    // 3. الترتيب
    const ranked = this.rankPosts(scored);
    
    // 4. التنويع
    const diversified = await this.diversifyFeed(ranked, userId);
    
    // 5. Pagination
    const start = (page - 1) * pageSize;
    const end = start + pageSize;
    
    return diversified.slice(start, end);
  }
}

export default new FeedAlgorithmService();
```

```typescript
interface UserInterests {
  carBrands: Map<string, number>;      // BMW: 15
  carTypes: Map<string, number>;       // SUV: 8
  priceRanges: Map<string, number>;    // 10k-20k: 12
  locations: Map<string, number>;      // Sofia: 20
  hashtags: Map<string, number>;       // #sport_cars: 7
  authors: Map<string, number>;        // userId: engagement
  postTypes: Map<string, number>;      // tip: 10
}

class PersonalizationService {
  // تتبع اهتمامات المستخدم
  async trackInterest(
    userId: string,
    action: UserAction
  ): Promise<void> {
    const interests = await this.getUserInterests(userId);
    
    switch (action.type) {
      case 'like':
      case 'love':
        await this.increaseInterest(interests, action.post, 3);
        break;
        
      case 'comment':
        await this.increaseInterest(interests, action.post, 5);
        break;
        
      case 'share':
        await this.increaseInterest(interests, action.post, 7);
        break;
        
      case 'save':
        await this.increaseInterest(interests, action.post, 6);
        break;
        
      case 'view':
        await this.increaseInterest(interests, action.post, 1);
        break;
        
      case 'skip':
        // Signal سلبي
        await this.decreaseInterest(interests, action.post, -2);
        break;
    }
    
    await this.saveUserInterests(userId, interests);
  }
  
  // زيادة الاهتمام
  private async increaseInterest(
    interests: UserInterests,
    post: Post,
    weight: number
  ) {
    // Car brand
    if (post.content.carReference) {
      const brand = post.content.carReference.brand;
      interests.carBrands.set(
        brand,
        (interests.carBrands.get(brand) || 0) + weight
      );
    }
    
    // Location
    if (post.location) {
      const city = post.location.city;
      interests.locations.set(
        city,
        (interests.locations.get(city) || 0) + weight
      );
    }
    
    // Hashtags
    post.content.hashtags?.forEach(tag => {
      interests.hashtags.set(
        tag,
        (interests.hashtags.get(tag) || 0) + weight
      );
    });
    
    // Author
    interests.authors.set(
      post.authorId,
      (interests.authors.get(post.authorId) || 0) + weight
    );
    
    // Post type
    interests.postTypes.set(
      post.type,
      (interests.postTypes.get(post.type) || 0) + weight
    );
  }
  
  // تخصيص النقاط
  async personalizeScore(
    post: Post,
    userId: string,
    baseScore: PostScore
  ): Promise<PostScore> {
    const interests = await this.getUserInterests(userId);
    let bonus = 0;
    
    // مكافأة حسب العلامة التجارية
    if (post.content.carReference) {
      const brandInterest = interests.carBrands.get(
        post.content.carReference.brand
      ) || 0;
      bonus += Math.min(brandInterest * 0.5, 10); // max +10
    }
    
    // مكافأة حسب المؤلف
    const authorInterest = interests.authors.get(post.authorId) || 0;
    bonus += Math.min(authorInterest * 0.3, 5); // max +5
    
    // مكافأة حسب الموقع
    if (post.location) {
      const locationInterest = interests.locations.get(
        post.location.city
      ) || 0;
      bonus += Math.min(locationInterest * 0.2, 3); // max +3
    }
    
    // مكافأة حسب Hashtags
    let hashtagBonus = 0;
    post.content.hashtags?.forEach(tag => {
      const tagInterest = interests.hashtags.get(tag) || 0;
      hashtagBonus += tagInterest * 0.1;
    });
    bonus += Math.min(hashtagBonus, 5); // max +5
    
    // مكافأة حسب نوع البوست
    const typeInterest = interests.postTypes.get(post.type) || 0;
    bonus += Math.min(typeInterest * 0.2, 3); // max +3
    
    return {
      ...baseScore,
      totalScore: Math.min(baseScore.totalScore + bonus, 100),
      personalizedBonus: bonus
    };
  }
  
  // Decay للاهتمامات القديمة
  async decayOldInterests(userId: string) {
    const interests = await this.getUserInterests(userId);
    const decayRate = 0.95; // 5% decay
    
    // تطبيق decay على جميع الاهتمامات
    interests.carBrands.forEach((value, key) => {
      interests.carBrands.set(key, value * decayRate);
    });
    
    interests.locations.forEach((value, key) => {
      interests.locations.set(key, value * decayRate);
    });
    
    // ... نفس الشيء للباقي
    
    await this.saveUserInterests(userId, interests);
  }
}

export default new PersonalizationService();
```

```typescript
// نموذج Exponential Decay
function calculateTimeDecay(ageHours: number): number {
  const halfLife = 6; // 6 ساعات
  return Math.exp(-0.693 * ageHours / halfLife);
}

// نموذج Linear Decay
function calculateLinearDecay(ageHours: number): number {
  const maxAge = 72; // 3 أيام
  return Math.max(0, 1 - (ageHours / maxAge));
}

// نموذج Step Decay (المستخدم في الخطة)
function calculateStepDecay(ageHours: number): number {
  if (ageHours < 1) return 1.0;
  if (ageHours < 3) return 0.9;
  if (ageHours < 6) return 0.75;
  if (ageHours < 12) return 0.6;
  if (ageHours < 24) return 0.4;
  if (ageHours < 72) return 0.2;
  return 0.05;
}
```

```typescript
class ViralDetector {
  async detectViral(postId: string): Promise<boolean> {
    const engagement = await this.getEngagementHistory(postId, 3600000); // آخر ساعة
    
    // حساب معدل النمو
    const growthRate = this.calculateGrowthRate(engagement);
    
    // Viral إذا:
    const isViral = 
      growthRate.likes > 10 && // 10 likes/hour
      growthRate.comments > 5 && // 5 comments/hour
      growthRate.shares > 3; // 3 shares/hour
    
    if (isViral) {
      await this.boostScore(postId, 15); // مكافأة +15
      await this.notifyFollowers(postId);
    }
    
    return isViral;
  }
  
  private calculateGrowthRate(history: EngagementHistory) {
    const hourAgo = Date.now() - 3600000;
    const recent = history.filter(e => e.timestamp > hourAgo);
    
    return {
      likes: recent.filter(e => e.type === 'like').length,
      comments: recent.filter(e => e.type === 'comment').length,
      shares: recent.filter(e => e.type === 'share').length
    };
  }
}
```
