// ReviewForm.tsx
import React, { useState } from 'react';
import './ReviewForm.css';

export default function ReviewForm({ onSubmit }) {
  const [name, setName] = useState('');
  const [rating, setRating] = useState(5);
  const [comment, setComment] = useState('');

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();
    if (name && comment) {
      onSubmit({ name, rating, comment });
      setName('');
      setRating(5);
      setComment('');
    }
  };

  return (
    <form className="review-form" onSubmit={handleSubmit}>
      <h3>📝 أضف تقييمك</h3>
      <input
        type="text"
        placeholder="اسمك"
        value={name}
        onChange={(e) => setName(e.target.value)}
        required
      />
      <select value={rating} onChange={(e) => setRating(Number(e.target.value))}>
        {[5, 4, 3, 2, 1].map((star) => (
          <option key={star} value={star}>{'⭐'.repeat(star)}</option>
        ))}
      </select>
      <textarea
        placeholder="اكتب تعليقك هنا..."
        value={comment}
        onChange={(e) => setComment(e.target.value)}
        required
      />
      <button type="submit">إرسال التقييم</button>
    </form>
  );
}

function calculateTrustScore(reviews) {
  if (reviews.length === 0) return 0;
  const total = reviews.reduce((sum, r) => sum + r.rating, 0);

}

tsx
import React, { useState, useEffect } from 'react';
import { db } from './firebase-config';
import { collection, addDoc, onSnapshot } from 'firebase/firestore';
import ReviewForm from './ReviewForm';
import './CustomerReviews.css';

export default function CustomerReviews() {
  const [reviews, setReviews] = useState([]);

  useEffect(() => {
    const unsubscribe = onSnapshot(collection(db, 'reviews'), (snapshot) => {
      const data = snapshot.docs.map(doc => doc.data());
      setReviews(data);
    });
    return () => unsubscribe();
  }, []);

  const handleNewReview = async (newReview) => {
    await addDoc(collection(db, 'reviews'), newReview);
  };

  const trustScore = calculateTrustScore(reviews);
  const isTrusted = trustScore >= 4.5 && reviews.length >= 10;

  return (
    <div className="reviews-section">
      <h2>⭐ تقييمات العملاء</h2>
      <div className="trust-score">
        <p>مستوى الثقة: <strong>{trustScore} / 5</strong> بناءً على <strong>{reviews.length}</strong> مراجعة</p>
        {isTrusted && <span className="badge">✅ بائع موثوق</span>}
      </div>

      <ReviewForm onSubmit={handleNewReview} />

      <div className="reviews-list">
        {reviews.map((review, index) => (
          <div key={index} className="review-card">
            <h4>{review.name}</h4>
            <p>{'⭐'.repeat(review.rating)}</p>
            <p>{review.comment}</p>
          </div>
        ))}
      </div>
    </div>
  );
}

function calculateTrustScore(reviews) {
  if (reviews.length === 0) return 0;
  const total = reviews.reduce((sum, r) => sum + r.rating, 0);
  return (total / reviews.length).toFixed(1);
}

🛡️ لوحة مراجعة التقييمات (Admin Review Panel)
✅ المكون في React
tsx
// ReviewAdminPanel.tsx
import React, { useEffect, useState } from 'react';
import { db } from './firebase-config';
import { collection, getDocs, deleteDoc, doc, updateDoc } from 'firebase/firestore';
import './ReviewAdminPanel.css';

export default function ReviewAdminPanel() {
  const [pendingReviews, setPendingReviews] = useState([]);

  useEffect(() => {
    const fetchReviews = async () => {
      const snapshot = await getDocs(collection(db, 'reviews'));
      const data = snapshot.docs
        .map(doc => ({ id: doc.id, ...doc.data() }))
        .filter(review => !review.approved);
      setPendingReviews(data);
    };
    fetchReviews();
  }, []);

  const approveReview = async (id) => {
    await updateDoc(doc(db, 'reviews', id), { approved: true });
    setPendingReviews(pendingReviews.filter(r => r.id !== id));
  };

  const rejectReview = async (id) => {
    await deleteDoc(doc(db, 'reviews', id));
    setPendingReviews(pendingReviews.filter(r => r.id !== id));
  };

  return (
    <div className="admin-panel">
      <h2>🛡️ مراجعة التقييمات قبل النشر</h2>
      {pendingReviews.length === 0 ? (
        <p>لا توجد تقييمات قيد المراجعة حاليًا.</p>
      ) : (
        pendingReviews.map((review) => (
          <div key={review.id} className="review-card">
            <h4>{review.name}</h4>
            <p>{'⭐'.repeat(review.rating)}</p>
            <p>{review.comment}</p>
            <div className="actions">
              <button onClick={() => approveReview(review.id)}>✅ موافقة</button>
              <button onClick={() => rejectReview(review.id)}>❌ رفض</button>
            </div>
          </div>
        ))
      )}
    </div>
  );
}

🛡️ نظام صلاحيات المستخدمين (Role-Based Access Control)
✅ نموذج الأدوار المقترحة
الدور	الصلاحيات
المدير (Admin)	الوصول الكامل: تعديل العروض، مراجعة التقييمات، إدارة الحملات، حذف الحسابات
البائع (Seller)	إضافة وتعديل المركبات، الرد على الرسائل، إنشاء حملات إعلانية
الزائر (Visitor)	عرض المركبات فقط، إرسال رسائل، مشاهدة التقييمات
🧠 كود التحقق من الدور داخل React
tsx
// useUserRole.ts
import { useEffect, useState } from 'react';
import { db } from './firebase-config';
import { doc, getDoc } from 'firebase/firestore';

export function useUserRole(userId) {
  const [role, setRole] = useState(null);

  useEffect(() => {
    const fetchRole = async () => {
      const docRef = doc(db, 'users', userId);
      const docSnap = await getDoc(docRef);
      if (docSnap.exists()) {
        setRole(docSnap.data().role); // "admin", "seller", "visitor"
      }
    };
    if (userId) fetchRole();
  }, [userId]);

  return role;
}

الخلاصة
التحليل يظهر أن الفريق أنشأ نظاماً اجتماعياً متكاملاً ومتقناً تقنياً في وقت قياسي (4 ساعات). النظام لا يضيف ميزات جديدة فحسب، بل يحول المنصة من سوق تقليدي إلى مجتمع نابض بالحياة، مما يخلق قيمة هائلة للمستخدمين والمنصة على حد سواء.

التنفيذ أظهر فهمًا عميقًا للتحديات التقنية (مشاكل قابلية التوسع، الأمان، الأداء) وقدم حلولاً عملية وفعالة. النظام الآن جاهز للإطلاق ويمثل نقلة نوعية في رحلة المنصة.

التأثير الاستراتيجي النهائي:
تم تحويل Globul Cars من سوق سيارات تقليدي إلى منصة اجتماعية متكاملة، مما يخلق دفاعاً تنافسياً قويًا ويضع الأساس لنمو أسي وقيمة استثمارية مضاعفة.

هذا الإنجاز يمثل تحولاً استراتيجياً كاملاً في نموذج العمل، وليس مجرد إضافة ميزات جديدة. البنية التحتية المبنية تسمح بتحقيق تأثير الشبكة (Network Effects) وخلق قيمة متزايدة مع كل مستخدم جديد.

الان اكتب استكمال للملفات ال3 التي حللتها بخطة رصينه تطويرية عميقه ليكون نظام تواصل اجتماعي متكامل من كل النواحي باضافة كل مقترحاتك الذكية لكن لا تعيد ما موجود في هذه الملفات وانما استتكمال لها و تزاوج  و تكامل مع المشروع الحالي وهذه الخطط ال3 الموجودات عندك لكن اكتب شيء كامل 100% وليس متثطع يعني في النهاية اكتب لي نفذ كل شيء ولا داعي ان اقول لك اكمل
نحن نريد استكمال النظام الاجتماعي بشكل كامل مع التكامل مع المشروع الحالي. لن نعيد ما تم تنفيذه، بل سنبني عليه.

1. نظام الرسائل الفورية (Real-time Messaging)
الهدف:
تمكين المستخدمين من إرسال رسائل فورية بعضهم إلى بعض.

التصميم:
مجموعة البيانات:
conversations: مجموعة للمحادثات بين مستخدمين أو أكثر.

id: معرف المحادثة.

participants: مصفوفة من معرفات المستخدمين المشاركين.

lastMessage: آخر رسالة (لتحديث قائمة المحادثات).

updatedAt: تاريخ آخر تحديث.

type: نوع المحادثة (خاص، جماعية، etc.)

conversations/{conversationId}/messages: مجموعة فرعية للرسائل.

id: معرف الرسالة.

senderId: مرسل الرسالة.

content: محتوى الرسالة.

timestamp: وقت الإرسال.

type: نوع الرسالة (نص، صورة، etc.)

status: حالة الرسالة (مرسلة، وصلت، مقروءة)

الخدمات:
messaging.service.ts: لإنشاء المحادثات، إرسال الرسائل، واسترجاعها.

المكونات:
ConversationList.tsx: قائمة المحادثات.

ChatWindow.tsx: نافذة المحادثة.

الدوال السحابية:
onNewMessage: لتحديث lastMessage في المحادثة وإرسال إشعارات.

2. نظام الإشعارات (Notifications)

التصميم:

notifications: مجموعة للإشعارات.

id: معرف الإشعار.

userId: المستخدم المستهدف.

type: نوع الإشعار (مثل: إعجاب، تعليق، رسالة، etc.)

relatedId: المعرف المرتبط (مثل: معرف المنشور)

message: نص الإشعار.

isRead: هل تم القراءة؟

createdAt: وقت الإنشاء.

الخدمات:
notifications.service.ts: لإنشاء الإشعارات، قراءتها، وحذفها.

المكونات:
NotificationsBell.tsx: زر الجرس لقائمة الإشعارات.

NotificationsList.tsx: قائمة الإشعارات.

الدوال السحابية:
onNotificationEvent: لإنشاء الإشعارات عند حدوث أنشطة (مثل: إعجاب، تعليق، etc.)

الخدمات:
