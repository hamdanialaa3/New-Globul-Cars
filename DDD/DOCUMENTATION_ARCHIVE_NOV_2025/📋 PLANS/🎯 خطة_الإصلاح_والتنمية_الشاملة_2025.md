النمط الموحد:
```typescript
const handleAction = async () => {
  try {
    setLoading(true);
    setError(null);
    
    const result = await someAsyncOperation();
    // Success handling

    const message = getErrorMessage(error);
    setError(message);
    showToast('error', message);
    logger.error('Operation failed', error as Error, { context });
    
  } finally {
    setLoading(false);
  }
};
```

معيار القبول:

  });
