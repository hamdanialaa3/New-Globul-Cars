## 4.1 Collaborative Filtering

### الملف: `src/services/social/ml/collaborative-filtering.service.ts` (250 سطر)

**المفهوم:** "المستخدمون المشابهون لك أعجبوا بهذا"

```typescript
class CollaborativeFilteringService {
  // إيجاد مستخدمين مشابهين
  async findSimilarUsers(userId: string, limit: number = 20) {
    const userEngagements = await this.getUserEngagements(userId);
    const allUsers = await this.getActiveUsers(1000);
    
    const similarities = allUsers.map(otherUser => {
      const similarity = this.calculateCosineSimilarity(
        userEngagements,
        otherUser.engagements
      );
      
      return { userId: otherUser.id, similarity };
    });
    
    return similarities
      .sort((a, b) => b.similarity - a.similarity)
      .slice(0, limit);
  }
  
  // حساب Cosine Similarity
  private calculateCosineSimilarity(
    vec1: Map<string, number>,
    vec2: Map<string, number>
  ): number {
    const keys = new Set([...vec1.keys(), ...vec2.keys()]);
    
    let dotProduct = 0;
    let norm1 = 0;
    let norm2 = 0;
    
    keys.forEach(key => {
      const v1 = vec1.get(key) || 0;
      const v2 = vec2.get(key) || 0;
      
      dotProduct += v1 * v2;
      norm1 += v1 * v1;
      norm2 += v2 * v2;
    });
    
    if (norm1 === 0 || norm2 === 0) return 0;
    
    return dotProduct / (Math.sqrt(norm1) * Math.sqrt(norm2));
  }
  
  // التوصية
  async recommendFromSimilarUsers(userId: string) {
    const similarUsers = await this.findSimilarUsers(userId, 20);
    const recommendations = [];
    
    for (const similar of similarUsers) {
      const theirLikedPosts = await this.getEngagedPosts(
        similar.userId,
        'like',
        50
      );
      
      recommendations.push(...theirLikedPosts.map(post => ({
        post,
        reason: 'similar_users',
        confidence: similar.similarity
      })));
    }
    
    return this.deduplicateAndScore(recommendations, userId);
  }
}
```

## 4.2 Content-Based Filtering

```typescript
class ContentFilteringService {
  // استخراج features
  extractFeatures(post: Post) {
    return {
      carBrand: post.content.carReference?.brand || '',
      carModel: post.content.carReference?.model || '',
      carType: post.content.carReference?.type || '',
      priceRange: this.getPriceRange(post.content.carReference?.price),
      year: post.content.carReference?.year || 0,
      location: post.location?.city || '',
      hashtags: new Set(post.content.hashtags || []),
      type: post.type,
      hasMedia: post.content.media ? 1 : 0,
      mediaType: post.content.media?.type || '',
      textLength: post.content.text.length,
      wordCount: post.content.text.split(/\s+/).length
    };
  }
  
  // حساب التشابه
  calculateSimilarity(post1: Post, post2: Post): number {
    const f1 = this.extractFeatures(post1);
    const f2 = this.extractFeatures(post2);
    
    let similarity = 0;
    
    // Car brand (30%)
    if (f1.carBrand && f1.carBrand === f2.carBrand) {
      similarity += 0.3;
    }
    
    // Location (20%)
    if (f1.location && f1.location === f2.location) {
      similarity += 0.2;
    }
    
    // Post type (15%)
    if (f1.type === f2.type) {
      similarity += 0.15;
    }
    
    // Hashtags (25%)
    const commonHashtags = [...f1.hashtags].filter(h => f2.hashtags.has(h));
    const hashtagSimilarity = commonHashtags.length / 
      Math.max(f1.hashtags.size, f2.hashtags.size, 1);
    similarity += hashtagSimilarity * 0.25;
    
    // Media type (10%)
    if (f1.hasMedia === f2.hasMedia && f1.mediaType === f2.mediaType) {
      similarity += 0.1;
    }
    
    return similarity;
  }
  
  // التوصية
  async recommendSimilarContent(userId: string) {
    const userLikedPosts = await this.getEngagedPosts(userId, 'like', 50);
    const candidatePosts = await this.getRecentPosts(200);
    
    const scored = candidatePosts.map(candidate => {
      const similarities = userLikedPosts.map(liked =>
        this.calculateSimilarity(liked, candidate)
      );
      
      const avgSimilarity = similarities.reduce((a, b) => a + b, 0) / 
        similarities.length;
      
      return { post: candidate, score: avgSimilarity };
    });
    
    return scored
      .sort((a, b) => b.score - a.score)
      .slice(0, 20);
  }
}
```

```typescript
class HybridRecommenderService {
  async getHybridRecommendations(
    userId: string,
    limit: number = 20
  ): Promise<Post[]> {
    // 1. Collaborative (40%)
    const collaborative = await collaborativeFilteringService
      .recommendFromSimilarUsers(userId);
    
    // 2. Content-Based (30%)
    const contentBased = await contentFilteringService
      .recommendSimilarContent(userId);
    
    // 3. Trending (20%)
    const trending = await this.getTrendingPosts(50);
    
    // 4. Fresh/Random (10%)
    const fresh = await this.getRandomRecentPosts(20);
    
    // دمج
    const combined = [
      ...collaborative.slice(0, Math.ceil(limit * 0.4)),
      ...contentBased.slice(0, Math.ceil(limit * 0.3)),
      ...trending.slice(0, Math.ceil(limit * 0.2)),
      ...fresh.slice(0, Math.ceil(limit * 0.1))
    ];
    
    // إزالة المكررات
    const unique = this.deduplicatePosts(combined);
    
    // Re-rank
    return this.rerankByFinalScore(unique, userId);
  }
  
  private async rerankByFinalScore(
    posts: Post[],
    userId: string
  ): Promise<Post[]> {
    const scored = await Promise.all(
      posts.map(async post => {
        const baseScore = await postScoringService.calculatePostScore(post.id);
        const personalizedScore = await personalizationService
          .personalizeScore(post, userId, baseScore);
        
        return { post, score: personalizedScore.totalScore };
      })
    );
    
    return scored
      .sort((a, b) => b.score - a.score)
      .map(s => s.post);
  }
}
```
