7. التنفيذ في المكون الرئيسي
tsx
// App.tsx أو الملف الرئيسي
import React, { useState } from 'react';
import CarSearchSystemAdvanced from './components/CarSearchSystemAdvanced';
import SearchResults from './components/SearchResults';

const App: React.FC = () => {
  const [searchResults, setSearchResults] = useState<CarDataFromFile[]>([]);
  const [searchParams, setSearchParams] = useState<any>({});

  const handleSearchResults = (results: CarDataFromFile[]) => {
    setSearchResults(results);
  };

  const handleSearchParamsChange = (params: any) => {
    setSearchParams(params);
  };

  return (
    <div className="app">
      <CarSearchSystemAdvanced
        onSearchResults={handleSearchResults}
        onSearchParamsChange={handleSearchParamsChange}
      />
      
      <SearchResults 
        results={searchResults}
        searchParams={searchParams}
      />
    </div>
  );
};

export default App;
