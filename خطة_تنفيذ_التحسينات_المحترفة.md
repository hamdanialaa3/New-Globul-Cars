// ✅ src/services/car-count.service.ts
/**
 * Car Count Service
 * 
 * Purpose:
 * - Get total count of cars across all collections
 * - Cached count for performance
 * - Real-time updates for accuracy
 * 
 * Architecture:
 * - Uses Multi-collection pattern (SellWorkflowCollections)
 * - Numeric ID compatible
 * - Firestore optimized queries
 */

import { db } from '@/services/firebase';
import { collection, getCountFromServer, query, where } from 'firebase/firestore';
import { logger } from '@/services/logger-service';
import { SellWorkflowCollections } from '@/services/sell-workflow-collections.service';

interface CarCountResult {
  total: number;
  byCollection: Record<string, number>;
  lastUpdated: Date;
}

class CarCountService {
  private static instance: CarCountService | null = null;
  private cache: CarCountResult | null = null;
  private cacheExpiry = 5 * 60 * 1000; // 5 minutes
  private lastFetchTime: number = 0;

  private constructor() {}

  public static getInstance(): CarCountService {
    if (!CarCountService.instance) {
      CarCountService.instance = new CarCountService();
    }
    return CarCountService.instance;
  }

  /**
   * Get total car count across all collections
   * Cached for 5 minutes for performance
   */
  async getTotalCount(forceRefresh: boolean = false): Promise<number> {
    try {
      const now = Date.now();
      
      // Return cached if still valid and not forced
      if (!forceRefresh && this.cache && (now - this.lastFetchTime < this.cacheExpiry)) {
        logger.debug('Car count from cache', { total: this.cache.total });
        return this.cache.total;
      }

      // Fetch from all collections
      const collections = [
        'passenger_cars',
        'suvs',
        'vans',
        'motorcycles',
        'trucks',
        'buses'
      ];

      const countPromises = collections.map(async (collectionName) => {
        try {
          const collectionRef = collection(db, collectionName);
          // Only count active, non-sold cars
          const activeQuery = query(
            collectionRef,
            where('isSold', '==', false),
            where('status', '==', 'active')
          );
          const snapshot = await getCountFromServer(activeQuery);
          return { collection: collectionName, count: snapshot.data().count };
        } catch (error) {
          logger.error(`Error counting ${collectionName}`, error as Error, { collection: collectionName });
          return { collection: collectionName, count: 0 };
        }
      });

      const results = await Promise.all(countPromises);
      
      const byCollection: Record<string, number> = {};
      let total = 0;
      
      results.forEach(({ collection, count }) => {
        byCollection[collection] = count;
        total += count;
      });

      // Update cache
      this.cache = {
        total,
        byCollection,
        lastUpdated: new Date()
      };
      this.lastFetchTime = now;

      logger.info('Car count fetched', { total, byCollection });
      return total;

    } catch (error) {
      logger.error('Error getting total car count', error as Error);
      // Return cached value if available, otherwise default
      return this.cache?.total ?? 0;
    }
  }

  /**
   * Get count by collection
   */
  async getCountByCollection(collectionName: string): Promise<number> {
    const counts = await this.getTotalCount();
    return this.cache?.byCollection[collectionName] ?? 0;
  }

  /**
   * Invalidate cache (call after car creation/deletion)
   */
  invalidateCache(): void {
    this.cache = null;
    this.lastFetchTime = 0;
    logger.debug('Car count cache invalidated');
  }
}

export const carCountService = CarCountService.getInstance();
```

import { logger } from '../services/logger-service';
import { carCountService } from '../services/car-count.service';

export const onCarCountChange = functions.firestore
  .document('{collection}/{carId}')
  .onWrite(async (change, context) => {
    try {
      const { collection } = context.params;
      
      // Only process vehicle collections
      const vehicleCollections = [
        'passenger_cars', 'suvs', 'vans', 
        'motorcycles', 'trucks', 'buses'
      ];
      
      if (!vehicleCollections.includes(collection)) {
        return null;
      }

      // Invalidate cache (frontend will refresh on next request)
      // Note: In a real implementation, you might want to update a cache document
      // in Firestore that frontend can listen to
      
      logger.info('Car count cache should be invalidated', {
        collection,
        carId: context.params.carId,
        eventType: change.after.exists ? 'create/update' : 'delete'
      });

      return null;
    } catch (error) {
      logger.error('Error in onCarCountChange', error as Error, { context });
      return null;
    }
  });

// ✅ src/components/Image/OptimizedImage.tsx
/**
 * Optimized Image Component
 * 
 * Features:
 * - WebP format (CONSTITUTION requirement)
 * - Lazy loading
 * - Blur placeholder
 * - Responsive sizes
 * - Error handling
 * 
 * Architecture:
 * - Uses browser-image-compression for WebP conversion
 * - Mobile-first responsive
 * - Error boundaries
 */

import React, { useState, useCallback } from 'react';
import styled from 'styled-components';
import { logger } from '@/services/logger-service';

interface OptimizedImageProps {
  src: string;
  alt: string;
  width?: number;
  height?: number;
  className?: string;
  loading?: 'lazy' | 'eager';
  priority?: boolean;
  sizes?: string;
  objectFit?: 'cover' | 'contain' | 'fill' | 'none' | 'scale-down';
}

const ImageContainer = styled.div<{ $width?: number; $height?: number }>`
  position: relative;
  width: ${props => props.$width ? `${props.$width}px` : '100%'};
  height: ${props => props.$height ? `${props.$height}px` : 'auto'};
  overflow: hidden;
  background: var(--bg-secondary);
`;

const StyledImage = styled.img<{ $objectFit?: string; $loaded: boolean }>`
  width: 100%;
  height: 100%;
  object-fit: ${props => props.$objectFit || 'cover'};
  transition: opacity 0.3s ease;
  opacity: ${props => props.$loaded ? 1 : 0};
`;

const BlurPlaceholder = styled.div`
  position: absolute;
  inset: 0;
  background: linear-gradient(
    90deg,
    var(--bg-secondary) 0%,
    var(--bg-tertiary) 50%,
    var(--bg-secondary) 100%
  );
  background-size: 200% 100%;
  animation: shimmer 1.5s infinite;
  
  @keyframes shimmer {
    0% { background-position: -200% 0; }
    100% { background-position: 200% 0; }
  }
`;

const ErrorPlaceholder = styled.div`
  position: absolute;
  inset: 0;
  display: flex;
  align-items: center;
  justify-content: center;
  background: var(--bg-secondary);
  color: var(--text-secondary);
  font-size: 0.875rem;
`;

export const OptimizedImage: React.FC<OptimizedImageProps> = ({
  src,
  alt,
  width,
  height,
  className,
  loading = 'lazy',
  priority = false,
  sizes = '(max-width: 768px) 100vw, (max-width: 1200px) 50vw, 300px',
  objectFit = 'cover'
}) => {
  const [isLoaded, setIsLoaded] = useState(false);
  const [hasError, setHasError] = useState(false);
  const [imageSrc, setImageSrc] = useState<string>('');

  // Convert to WebP if needed
  useEffect(() => {
    const loadImage = async () => {
      try {
        // If already WebP, use directly
        if (src.endsWith('.webp') || src.includes('format=webp')) {
          setImageSrc(src);
          return;
        }

        // For external images, try to use Cloud Function conversion
        // or return original if conversion not available
        setImageSrc(src);
      } catch (error) {
        logger.error('Error processing image', error as Error, { src });
        setImageSrc(src); // Fallback to original
      }
    };

    loadImage();
  }, [src]);

  const handleLoad = useCallback(() => {
    setIsLoaded(true);
  }, []);

  const handleError = useCallback((error: React.SyntheticEvent<HTMLImageElement, Event>) => {
    logger.error('Image load error', new Error('Image failed to load'), {
      src: imageSrc,
      alt
    });
    setHasError(true);
    setIsLoaded(false);
  }, [imageSrc, alt]);

  if (hasError) {
    return (
      <ImageContainer $width={width} $height={height} className={className}>
        <ErrorPlaceholder>
          {alt || 'Image'}
        </ErrorPlaceholder>
      </ImageContainer>
    );
  }

  return (
    <ImageContainer $width={width} $height={height} className={className}>
      {!isLoaded && <BlurPlaceholder />}
      <StyledImage
        src={imageSrc || src}
        alt={alt}
        loading={priority ? 'eager' : loading}
        decoding="async"
        $objectFit={objectFit}
        $loaded={isLoaded}
        onLoad={handleLoad}
        onError={handleError}
        width={width}
        height={height}
        sizes={sizes}
      />
    </ImageContainer>
  );
};
```

### ✅ Task 2.1: Auto-complete Search

```typescript
// ✅ src/components/Search/BrandAutocomplete.tsx

/**
 * Brand Autocomplete Component
 * 
 * Features:
 * - Real-time suggestions
 * - Keyboard navigation
 * - Mobile-optimized
 * - Accessibility compliant
 * 
 * Architecture:
 * - Uses brandsModelsDataService
 * - Debounced input
 * - Virtualized list for performance
 */

import React, { useState, useEffect, useRef, useCallback } from 'react';
import { useLanguage } from '@/contexts/LanguageContext';
import { brandsModelsDataService } from '@/services/brands-models-data.service';
import { logger } from '@/services/logger-service';
import { sanitizeCarMakeModel } from '@/utils/inputSanitizer';
// ... styled components ...

export const BrandAutocomplete: React.FC<{
  value: string;
  onChange: (value: string) => void;
  onSelect: (value: string) => void;
}> = ({ value, onChange, onSelect }) => {
  const { language } = useLanguage();
  const [suggestions, setSuggestions] = useState<string[]>([]);
  const [isOpen, setIsOpen] = useState(false);
  const [highlightedIndex, setHighlightedIndex] = useState(-1);
  const inputRef = useRef<HTMLInputElement>(null);
  const listRef = useRef<HTMLUListElement>(null);
  const [brands, setBrands] = useState<string[]>([]);

  // Load brands on mount
  useEffect(() => {
    let isActive = true;

    const loadBrands = async () => {
      try {
        const allBrands = await brandsModelsDataService.getAllBrands();
        if (isActive) {
          setBrands(allBrands);
        }
      } catch (error) {
        logger.error('Error loading brands for autocomplete', error as Error);
      }
    };

    loadBrands();

    return () => {
      isActive = false;
    };
  }, []);

  // Filter suggestions
  useEffect(() => {
    if (!value || value.length < 2) {
      setSuggestions([]);
      setIsOpen(false);
      return;
    }

    const sanitized = sanitizeCarMakeModel(value);
    const filtered = brands
      .filter(brand => 
        brand.toLowerCase().includes(sanitized.toLowerCase())
      )
      .slice(0, 10); // Limit to 10 suggestions

    setSuggestions(filtered);
    setIsOpen(filtered.length > 0);
    setHighlightedIndex(-1);
  }, [value, brands]);

  // Keyboard navigation
  const handleKeyDown = useCallback((e: React.KeyboardEvent) => {
    if (!isOpen || suggestions.length === 0) {
      if (e.key === 'ArrowDown') {
        e.preventDefault();
        setIsOpen(true);
      }
      return;
    }

    switch (e.key) {
      case 'ArrowDown':
        e.preventDefault();
        setHighlightedIndex(prev => 
          prev < suggestions.length - 1 ? prev + 1 : prev
        );
        break;
      case 'ArrowUp':
        e.preventDefault();
        setHighlightedIndex(prev => prev > 0 ? prev - 1 : -1);
        break;
      case 'Enter':
        e.preventDefault();
        if (highlightedIndex >= 0 && highlightedIndex < suggestions.length) {
          const selected = suggestions[highlightedIndex];
          onSelect(selected);
          setIsOpen(false);
        }
        break;
      case 'Escape':
        setIsOpen(false);
        setHighlightedIndex(-1);
        break;
    }
  }, [isOpen, suggestions, highlightedIndex, onSelect]);

  // Scroll highlighted item into view
  useEffect(() => {
    if (highlightedIndex >= 0 && listRef.current) {
      const items = listRef.current.children;
      if (items[highlightedIndex]) {
        items[highlightedIndex].scrollIntoView({
          block: 'nearest',
          behavior: 'smooth'
        });
      }
    }
  }, [highlightedIndex]);

  const handleSelect = useCallback((brand: string) => {
    onSelect(brand);
    setIsOpen(false);
    inputRef.current?.focus();
  }, [onSelect]);

  return (
    <AutocompleteContainer>
      <Input
        ref={inputRef}
        type="text"
        value={value}
        onChange={(e) => onChange(sanitizeCarMakeModel(e.target.value))}
        onKeyDown={handleKeyDown}
        onFocus={() => {
          if (suggestions.length > 0) {
            setIsOpen(true);
          }
        }}
        onBlur={(e) => {
          // Delay to allow click events
          setTimeout(() => {
            if (!e.currentTarget.contains(document.activeElement)) {
              setIsOpen(false);
            }
          }, 200);
        }}
        aria-label={language === 'bg' ? 'Търси марка' : 'Search brand'}
        aria-autocomplete="list"
        aria-expanded={isOpen}
        aria-controls="brand-suggestions"
        role="combobox"
      />
      
      {isOpen && suggestions.length > 0 && (
        <SuggestionsList
          ref={listRef}
          id="brand-suggestions"
          role="listbox"
          aria-label={language === 'bg' ? 'Предложения' : 'Suggestions'}
        >
          {suggestions.map((brand, index) => (
            <SuggestionItem
              key={brand}
              $highlighted={index === highlightedIndex}
              onClick={() => handleSelect(brand)}
              onMouseEnter={() => setHighlightedIndex(index)}
              role="option"
              aria-selected={index === highlightedIndex}
            >
              {brand}
            </SuggestionItem>
          ))}
        </SuggestionsList>
      )}
    </AutocompleteContainer>
  );
};
```
