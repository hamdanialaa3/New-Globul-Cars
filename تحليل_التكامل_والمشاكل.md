```typescript
// ✅ تحديث getPermissions لتطابق الخطط الحالية
function getPermissions(profileType: ProfileType, planTier: PlanTier): ProfilePermissions {
  const PLAN_LIMITS: Record<PlanTier, number> = {
    free: 5,
    dealer: 15,
    company: -1
  };

  const maxListings = PLAN_LIMITS[planTier] || 5;

  const base: ProfilePermissions = {
    canAddListings: true,
    maxListings,
    hasAnalytics: false,
    hasAdvancedAnalytics: false,
    hasTeam: false,
    canExportData: false,
    hasPrioritySupport: false,
    canUseQuickReplies: false,
    canBulkEdit: false,
    canImportCSV: false,
    canUseAPI: false
  };

  // ✅ Dealer plan
  if (profileType === 'dealer' && planTier === 'dealer') {
    return {
      ...base,
      hasAnalytics: true,
      canUseQuickReplies: true,
      canBulkEdit: true,
      hasPrioritySupport: true
    };
  }

  // ✅ Company plan
  if (profileType === 'company' && planTier === 'company') {
    return {
      ...base,
      hasAnalytics: true,
      hasAdvancedAnalytics: true,
      hasTeam: true,
      canExportData: true,
      canUseQuickReplies: true,
      canBulkEdit: true,
      canImportCSV: true,
      canUseAPI: true,
      hasPrioritySupport: true
    };
  }

  return base;
}
```
