class CarDataBrowserService {
  private carData: CarDataFromFile[] = [];
  private summary: CarDataSummary | null = null;

  async loadData(): Promise<void> {
    try {
      const [carDataResponse, summaryResponse] = await Promise.all([
        fetch('/data/car-data.json'),
        fetch('/data/car-data-summary.json')
      ]);
      
      this.carData = await carDataResponse.json();
      this.summary = await summaryResponse.json();
      
      this.processData();
    } catch (error) {
      console.error('Error loading car data:', error);
      throw error;
    }
  }

  advancedSearch(filters: SearchParams): CarDataFromFile[] {
    let results = [...this.carData];

    // Basic Data Filters
    if (filters.make) {
      results = results.filter(car => 
        car.brand.toLowerCase() === filters.make.toLowerCase()
      );
    }

    if (filters.model) {
      results = results.filter(car => 
        car.model.toLowerCase().includes(filters.model.toLowerCase())
      );
    }

    if (filters.vehicleType.length > 0) {
      results = results.filter(car =>
        filters.vehicleType.includes(this.mapVehicleType(car.category))
      );
    }

    // Price Filter
    if (filters.minPrice) {
      const minPrice = parseFloat(filters.minPrice);
      results = results.filter(car => car.price >= minPrice);
    }

    if (filters.maxPrice) {
      const maxPrice = parseFloat(filters.maxPrice);
      results = results.filter(car => car.price <= maxPrice);
    }

    // Year Filter
    if (filters.minFirstRegistration) {
      const minYear = parseInt(filters.minFirstRegistration);
      results = results.filter(car => car.year >= minYear);
    }

    if (filters.maxFirstRegistration) {
      const maxYear = parseInt(filters.maxFirstRegistration);
      results = results.filter(car => car.year <= maxYear);
    }

    // Mileage Filter
    if (filters.minMileage) {
      const minMileage = parseInt(filters.minMileage);
      results = results.filter(car => car.mileage >= minMileage);
    }

    if (filters.maxMileage) {
      const maxMileage = parseInt(filters.maxMileage);
      results = results.filter(car => car.mileage <= maxMileage);
    }

    // Fuel Type Filter
    if (filters.fuelType.length > 0) {
      results = results.filter(car =>
        filters.fuelType.includes(car.fuelType)
      );
    }

    // Transmission Filter
    if (filters.transmission.length > 0) {
      results = results.filter(car =>
        filters.transmission.includes(car.transmission)
      );
    }

    // Location Filter
    if (filters.city) {
      results = results.filter(car =>
        car.location.toLowerCase().includes(filters.city.toLowerCase())
      );
    }

    // Search in Description
    if (filters.searchInDescription) {
      const searchTerm = filters.searchInDescription.toLowerCase();
      results = results.filter(car =>
        car.description?.toLowerCase().includes(searchTerm) ||
        car.features?.some(feature => feature.toLowerCase().includes(searchTerm))
      );
    }

    return results;
  }

  private mapVehicleType(category: string): string {
    const typeMap: { [key: string]: string } = {
      'седан': 'saloon',
      'комби': 'estate',
      'джип': 'suv',
      'купе': 'coupe',
      'кабрио': 'cabrio',
      'миниван': 'van'
    };
    return typeMap[category.toLowerCase()] || 'other';
  }

  getAllBrands(): string[] {
    return [...new Set(this.carData.map(car => car.brand))].sort();
  }

  getModelsByBrand(brand: string): string[] {
    return [...new Set(this.carData
      .filter(car => car.brand === brand)
      .map(car => car.model)
    )].sort();
  }

  getAvailableOptions(): any {
    return {
      vehicleTypes: ['седан', 'комби', 'джип', 'купе', 'кабрио', 'миниван'],
      fuelTypes: [...new Set(this.carData.map(car => car.fuelType))],
      transmissions: [...new Set(this.carData.map(car => car.transmission))],
      conditions: ['нов', 'употребяван', 'пререгистрация', 'служителска'],
      cities: ['София', 'Пловдив', 'Варна', 'Бургас', 'Русе', 'Стара Загора']
    };
  }
}

export const carDataBrowserService = new CarDataBrowserService();
