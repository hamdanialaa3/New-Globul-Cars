// --- Main Component: GlobalAuctions ---
export const GlobalAuctions: React.FC = () => {
    // State
    const [lang, setLang] = useState<Lang>('en');
    const [searchTerm, setSearchTerm] = useState('');
    const [isChatOpen, setIsChatOpen] = useState(false);
    const [apiKey, setApiKey] = useState('');
    const [messages, setMessages] = useState<{ text: string; sender: 'user' | 'bot'; isHtml?: boolean }[]>([
        { text: "Hello! I am your AI Auction Assistant ✨. Ask me about auction terms, repair estimates, or import rules to Bulgaria!", sender: 'bot' }
    ]);
    const [chatInput, setChatInput] = useState('');
    const [isLoading, setIsLoading] = useState(false);

    // Stats State
    const [liveAuctions, setLiveAuctions] = useState(4285);
    const [soldToday, setSoldToday] = useState(15492);
    const [activeBidders, setActiveBidders] = useState(124050);

    const chatContainerRef = useRef<HTMLDivElement>(null);
    const t = translations[lang];

    // Scroll to bottom of chat
    useEffect(() => {
        if (chatContainerRef.current) {
            chatContainerRef.current.scrollTop = chatContainerRef.current.scrollHeight;
        }
    }, [messages, isChatOpen]);

    // Live Stats Simulation
    useEffect(() => {
        const interval = setInterval(() => {
            if (Math.random() > 0.5) {
                setSoldToday(prev => prev + Math.floor(Math.random() * 3));
            }
            setLiveAuctions(prev => prev + (Math.floor(Math.random() * 10) - 5));
            setActiveBidders(prev => prev + (Math.floor(Math.random() * 50) - 20));
        }, 3000);

        return () => clearInterval(interval);
    }, []);

    // Interactive Card Effect
    const handleMouseMove = (e: React.MouseEvent<HTMLDivElement>) => {
        const card = e.currentTarget;
        const rect = card.getBoundingClientRect();
        const x = e.clientX - rect.left;
        const y = e.clientY - rect.top;
        card.style.setProperty("--mouse-x", `${x}px`);
        card.style.setProperty("--mouse-y", `${y}px`);
    };

    // Filter Logic
    const filteredAuctions = AUCTIONS_DATA.filter(item => {
        const titleMatch = item.title.toLowerCase().includes(searchTerm.toLowerCase());
        const descMatch = t[item.descKey]?.toLowerCase().includes(searchTerm.toLowerCase());
        const regionMatch = item.region.toLowerCase().includes(searchTerm.toLowerCase());
        return titleMatch || descMatch || regionMatch;
    });

    const renderSection = (regionCode: string, titleKey: string, flagUrl: string) => {
        const regionItems = filteredAuctions.filter(item => item.region === regionCode);
        if (regionItems.length === 0) return null;

        return (
            <section className="market-section">
                <div className="section-header">
                    <img src={flagUrl} alt="Region Flag" className="section-flag-icon" />
                    <h2>{t[titleKey] || titleKey}</h2>
                </div>
                <div className="auctions-grid">
                    {regionItems.map(item => (
                        <AuctionCard 
                            key={item.id}
                            flagBg={item.flagBg}
                            badge={item.badge}
                            tag={t[item.tagKey] || item.tagKey}
                            tagClass={item.tagClass}
                            title={item.title}
                            desc={t[item.descKey] || item.descKey}
                            link={item.link}
                            btnFlag={item.btnFlag}
                            btnText={t.btn_visit}
                            onMouseMove={handleMouseMove}
                        />
                    ))}
                </div>
            </section>
        );
    };

    // Chat Functions
    const callGemini = async (userPrompt: string): Promise<string> => {
        if (!apiKey) {
            return "Please enter a valid Gemini API Key in the settings field above to start.";
        }

        const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;
        
        const systemPrompt = "You are an expert automotive consultant for 'AutoPort BG', a global car auction portal. You help users understand car auctions (Copart, IAAI, Mobile.de, etc.), estimate repair costs based on descriptions (give ranges in EUR/BGN), and explain import regulations to Bulgaria. Be concise, professional, and helpful. Use emojis occasionally. If asked about prices, give estimated market ranges.";

        const payload = {
            contents: [{ parts: [{ text: userPrompt }] }],
            systemInstruction: { parts: [{ text: systemPrompt }] }
        };

        const response = await fetch(url, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(payload)
        });

        if (!response.ok) throw new Error('Gemini API Error');
        const data = await response.json();
        return data.candidates[0].content.parts[0].text;
    };

    const handleSendMessage = async () => {
        if (!chatInput.trim()) return;
        const userText = chatInput;
        setMessages(prev => [...prev, { text: userText, sender: 'user' }]);
        setChatInput('');
        setIsLoading(true);

        try {
            const response = await callGemini(userText);
            const htmlContent = await marked.parse(response);
            setMessages(prev => [...prev, { text: htmlContent, sender: 'bot', isHtml: true }]);
        } catch (error) {
            setMessages(prev => [...prev, { text: "Error: Please check your API Key or try again.", sender: 'bot' }]);
            console.error(error);
        } finally {
            setIsLoading(false);
        }
    };

    const handleKeyPress = (e: React.KeyboardEvent) => {
        if (e.key === 'Enter') handleSendMessage();
    };

    return (
        <div className="auctions-wrapper">
            <style>{styles}</style>
            <div className="auctions-container">
                {/* Nav */}
                <nav className="auctions-nav">
                    <div className="auctions-logo">Auto<span style={{color:'var(--accent-blue)'}}>Port</span> BG</div>
                    <div className="lang-switch">
                        <button className={`lang-btn ${lang === 'en' ? 'active' : ''}`} onClick={() => setLang('en')}>EN</button>
                        <button className={`lang-btn ${lang === 'bg' ? 'active' : ''}`} onClick={() => setLang('bg')}>BG</button>
                    </div>
                </nav>

                {/* Header */}
                <header className="auctions-header">
                    <h1>{t.header_title}</h1>
                    <p>{t.header_sub}</p>
                </header>

                {/* Search Bar (New Feature) */}
                <div className="search-container">
                    <div className="search-wrapper">
                        <i className="fa-solid fa-search search-icon"></i>
                        <input 
                            type="text" 
                            className="search-input" 
                            placeholder={t.search_placeholder}
                            value={searchTerm}
                            onChange={(e) => setSearchTerm(e.target.value)}
                        />
                    </div>
                </div>

                {/* Live Stats */}
                <div className="live-stats-container">
                    <div className="stats-bar">
                        <div className="stat-item">
                            <span className="live-dot"></span>
                            <div className="stat-content">
                                <span className="stat-number">{liveAuctions.toLocaleString()}</span>
                                <span className="stat-label">{t.stat_live}</span>
                            </div>
                        </div>
                        <div className="stat-item">
                            <i className="fa-solid fa-gavel stat-icon"></i>
                            <div className="stat-content">
                                <span className="stat-number">{soldToday.toLocaleString()}</span>
                                <span className="stat-label">{t.stat_sold}</span>
                            </div>
                        </div>
                        <div className="stat-item">
                            <i className="fa-solid fa-users stat-icon"></i>
                            <div className="stat-content">
                                <span className="stat-number">{activeBidders.toLocaleString()}</span>
                                <span className="stat-label">{t.stat_bidders}</span>
                            </div>
                        </div>
                        <div className="stat-item">
                            <i className="fa-solid fa-ship stat-icon"></i>
                            <div className="stat-content">
                                <span className="stat-number">8,730</span>
                                <span className="stat-label">{t.stat_transit}</span>
                            </div>
                        </div>
                    </div>
                </div>

                {/* Dynamic Sections */}
                {renderSection('usa', 'usa_title', 'https://flagcdn.com/w80/us.png')}
                {renderSection('de', 'de_title', 'https://flagcdn.com/w80/de.png')}
                {renderSection('eu', 'eu_title', 'https://flagcdn.com/w80/eu.png')}
                {renderSection('jp', 'jp_title', 'https://flagcdn.com/w80/jp.png')}
                {renderSection('global', 'global_title', 'https://flagcdn.com/w80/kr.png')}

                {/* Empty State */}
                {filteredAuctions.length === 0 && (
                    <div style={{textAlign: 'center', padding: '50px', color: 'var(--text-muted)'}}>
                        <i className="fa-solid fa-ghost" style={{fontSize: '3rem', marginBottom: '15px'}}></i>
                        <p>No auctions found matching "{searchTerm}"</p>
                    </div>
                )}

                {/* Footer */}
                <footer className="auctions-footer">
                    <p>&copy; 2024 AutoPort BG. {t.footer_rights}</p>
                </footer>
            </div>

            {/* Widgets (Viber & AI) */}
            <div className="widgets-container">
                {/* Viber Button (New Feature) */}
                <a href="viber://chat?number=%2B359000000000" className="viber-fab" title="Chat on Viber">
                    <i className="fa-brands fa-viber"></i>
                </a>

                {/* AI Button */}
                <div 
                    className="ai-fab" 
                    onClick={() => setIsChatOpen(true)}
                    style={{ transform: isChatOpen ? 'scale(0)' : 'scale(1)' }}
                    title="AI Assistant"
                >
                    <i className="fa-solid fa-wand-magic-sparkles"></i>
                </div>
            </div>

            <div className={`ai-chat-window ${isChatOpen ? 'active' : ''}`}>
                <div className="chat-header">
                    <span className="chat-title"><i className="fa-solid fa-robot"></i> AutoPort AI Advisor</span>
                    <button className="close-chat" onClick={() => setIsChatOpen(false)}><i className="fa-solid fa-xmark"></i></button>
                </div>

                {!apiKey && (
                    <div className="api-key-container" style={{display: 'block'}}>
                        <label>API Key Required (Gemini):</label>
                        <input 
                            type="password" 
                            className="api-key-input" 
                            placeholder="Paste your Gemini API Key here..."
                            onChange={(e) => setApiKey(e.target.value)}
                            value={apiKey}
                        />
                    </div>
                )}

                <div className="chat-messages" ref={chatContainerRef}>
                    {messages.map((msg, idx) => (
                        <div key={idx} className={`message ${msg.sender}`}>
                            {msg.isHtml ? <span dangerouslySetInnerHTML={{ __html: msg.text }} /> : msg.text}
                        </div>
                    ))}
                    {isLoading && <div className="message bot"><i className="fa-solid fa-spinner fa-spin"></i> Thinking...</div>}
                </div>

                <div className="chat-input-area">
                    <input 
                        type="text" 
                        className="chat-input" 
                        placeholder="Ask about a car or auction term..."
                        value={chatInput}
                        onChange={(e) => setChatInput(e.target.value)}
                        onKeyPress={handleKeyPress}
                    />
                    <button className="send-btn" onClick={handleSendMessage}><i className="fa-solid fa-paper-plane"></i></button>
                </div>
            </div>
        </div>
    );
};

export default GlobalAuctions;
