## 📊 بنية البيانات

### إعلان السيارة (مستند Firestore)
```typescript
{
  // معلومات أساسية
  id: string;
  vehicleType: string;
  make: string;
  model: string;
  year: number;
  mileage: number;
  fuelType: string;
  transmission: string;
  
  // المعدات
  safetyEquipment: string[];
  comfortEquipment: string[];
  infotainmentEquipment: string[];
  extras: string[];
  
  // الصور
  images: string[]; // روابط Firebase Storage
  
  // التسعير
  price: number;
  currency: string; // EUR
  priceType: string;
  negotiable: boolean;
  financing: boolean;
  tradeIn: boolean;
  warranty: boolean;
  paymentMethods: string[];
  
  // الموقع (بنية محسنة)
  city: string; // للتوافق القديم
  region: string;
  locationData: {
    cityId: string;
    cityName: { en, bg, ar };
    coordinates: { lat, lng };
  };
  
  // البائع
  sellerType: string;
  sellerName: string;
  sellerEmail: string;
  sellerPhone: string;
  sellerId: string; // Firebase UID
  
  // البيانات الوصفية
  status: 'draft' | 'active' | 'sold' | 'expired';
  isActive: boolean;
  isSold: boolean;
  isFeatured: boolean;
  views: number;
  favorites: number;
  createdAt: Timestamp;
  updatedAt: Timestamp;
}
```
